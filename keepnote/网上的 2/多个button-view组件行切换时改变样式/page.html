        {{item}}
      </button>

wxjs代码：
onchangeTab: function(e){
    var index = 0;
    switch(e.target.id){
      case "btn0":
        index=0;
      break;
      case "btn1":
        index = 1;
        break;
      case "btn2":
        index = 2;
        break;
      case "btn3":
        index = 3;
        break;
      default:
        break;
    }
    this.setData({
      tabIndex:index
    })
  },


先是设置了button的id然后根据不同的id来区分点击的button
