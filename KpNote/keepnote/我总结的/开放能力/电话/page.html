// 判断用户输入的手机号
const inputPhoneNumber = function(e) {
  var ok = false
  if (11 === e.detail.value.phoneNumber.length) {
    // 判断用户输入的手机号码格式是否正确： 2019年
    if (!(/^1(3|4|5|6|7|8|9)\d{9}$/.test(e.detail.value.phoneNumber))) {
      console.log('手机号码有误，请重填')
    } else {
      ok = true
    }
  }
  console.log('ok=', ok)

  if (true === ok) {
    wx.showToast({
      title: '您已成功预约！',
    })
    console.log('您输入的手机号码是：', e.detail.value.phoneNumber)
  } else {
    wx.showToast({
      title: '手机号码有误',
    })
  }
}
