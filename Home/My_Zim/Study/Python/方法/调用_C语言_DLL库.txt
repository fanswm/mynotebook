====== 调用 C语言 DLL库 ======
创建日期 星期六 11 一月 2020

__/*hello.c*/__

#include <stdio.h>
void hello(){
		printf("Hello void by C!\n");
}

__# test.py__

#!/usr/bin/env python3
# coding:utf-8

from ctypes import *

mycall = CDLL('./hello.dll')
mycall.hello()	
#file test.py END.

__制作DLL文件__
gcc -c hello.c	#得到 hello.o
gcc -shared -o hello.dll hello.o	# 得到 hello.dll文件
